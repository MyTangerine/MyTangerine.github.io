对于java的单继承的弥补

对于java的单继承缺陷，我们可以使用java里面的接口与继承来弥补java的单继承缺陷。 因为继承是可以使用这样的格式的 interface Intername extends Intername,Inername 所以我们只要写好了两个接口，就可以在第三个接口里面同时继承另外两个接口。

package com.NewStart.gogo;

interface Inter1

{

public static final int AGE = 100;

public abstract void tell();

}

interface Inter2{

public abstract void say();

}

abstract class ClassInter3{

public abstract void print();

}

class A extends ClassInter3 implements Inter1,Inter2{

public void tell(){

}

public void say(){

}

public void print(){

}

}

interface Inter3 extends Inter2,Inter1 {

}

public class InterDemo01 {

public static void main(String[] args)

{

    A a =new A();

    a.tell();

    System.out.println(Inter1.AGE);

    a.print();

}

}
